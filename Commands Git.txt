mkdir - створення папки;

git init - ініціалізація репозиторія;

nano - створення та редагування файлу;

git add_назва файлу - індексація файлу на комміт кандидат;

git add . - індексація всіх файлів в поточній директорії на 

комміт кандидат;

git add * - індексація всіх untracked файлів на комміт кандидат;

gitk - вбудований редактор;

git status - перевірка стану репозиторія;

git commit -m "коментар" - запис змін в репозиторій;

git log - історія коммітів;

git log --pretty=oneline - виведення історії в одну строку;

                         --max-count=2 -виведення вказаної к-сті 
                         --author="" -виведення історії автора;

mkdir - створення папки;

git init - ініціалізація репозиторія;

nano - створення та редагування файлу;

git add_назва файлу - індексація файлу на комміт кандидат;

git add . - індексація всіх файлів в поточній директорії на 

комміт кандидат;

git add * - індексація всіх untracked файлів на комміт кандидат;

gitk - вбудований редактор;

git status - перевірка стану репозиторія;

git commit -m "коментар" - запис змін в репозиторій;

git log - історія коммітів;

git log --pretty=oneline - виведення історії в одну строку;

                         --max-count=2 -виведення вказаної к-сті 
                         --author="" -виведення історії автора;

git checkout - відміна змін у файлі(переміщення вказівника до конкретного комміту);

git reset head - відміна індексації;

git revert head - відміна комміта;

git checkout -b - створення гілки;

git merge - злиття гілок;

git clone - клонування репозиторію;

git push - завантаження локальних змін на віддалений репозиторій;

git stash - створення стешу;

git stash list - список стешів;

git stash apply - застосування стешу; 

git stash pop - застосування стешу та видалення його із списку;

git stash clear - видалення всіх стешів;

git stash drop - видалення конкретного стешу;
